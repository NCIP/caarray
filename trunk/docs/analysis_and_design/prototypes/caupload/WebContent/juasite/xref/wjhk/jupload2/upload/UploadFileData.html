153     /**
154      * This methods writes the file data (see {@link FileData#getInputStream()}
155      * to the given outputStream (the output toward the HTTP server).
156      * 
157      * @param outputStream The stream on which the data is to be written.
158      * @param amount The number of bytes to write.
159      * @throws JUploadException if an I/O error occurs.
160      * @throws JUploadInterrupted Thrown when an interruption of the thread is
161      *             detected.
162      */
163     void uploadFile(OutputStream outputStream, long amount)
164             throws JUploadException, JUploadInterrupted {
165         if (this.uploadPolicy.getDebugLevel()>=30) {
166             this.uploadPolicy.displayDebug("in UploadFileData.uploadFile (amount:"
167                     + amount + ", getUploadLength(): " + getUploadLength() + ")",
168                     30);
169         }
170 
171         // getInputStream will put a new fileInput in the inputStream attribute,
172         // or leave it unchanged if it is not null.
173         InputStream inputStream = getInputStream();
174 
175         while (amount > 0 && !this.fileUploadManagerThread.isUploadFinished()) {
176             // Are we interrupted ?
177             if (Thread.interrupted()) {
178                 throw new JUploadInterrupted(getClass().getName()
179                         + ".uploadFile [" + this.getFileName() + "]",
180                         this.uploadPolicy);
181             }
182 
183             int toread = (amount > BUFLEN) ? BUFLEN : (int) amount;
184             int towrite = 0;
185 
186             try {
187                 towrite = inputStream.read(this.readBuffer, 0, toread);
188             } catch (IOException e) {
189                 throw new JUploadIOException(e);
190             }
191             if (towrite > 0) {
192                 try {
193                     outputStream.write(this.readBuffer, 0, towrite);
194                     this.fileUploadManagerThread.nbBytesUploaded(towrite, this);
195                     amount -= towrite;
196                     this.uploadRemainingLength -= towrite;
197 
198                     // For debug reason, I may need to simulate upload, that are
199                     // on a real network. We then slow down the upload. This can
200                     // occurs only when given a 'high' debugLevel (higher than
201                     // what can be set with the applet GUI.
202                     if (this.uploadPolicy.getDebugLevel() > 100) {
203                         try {
204                             Thread.sleep(20);
205                         } catch (InterruptedException e) {
206                             // Nothing to do. We'll just take a look at the loop
207                             // condition.
208                         }
209                     }
210                 } catch (IOException ioe) {
211                     throw new JUploadIOException(this.getClass().getName()
212                             + ".uploadFile()", ioe);
213                 } catch (Exception e) {
214                     // When the user may not override an existing file, I got a
215                     // NullPointerException. Let's trap all errors here.
216                     throw new JUploadException(
217                             this.getClass().getName()
218                                     + ".uploadFile()  (check the user permission on the server)",
219                             e);
220                 }
221             }
222         }// while
223     }

225     /**
226      * Clean any local resources, then transmit to the encapsulated
227      * {@link FileData#afterUpload()}.
228      * 
229      * @see FileData#afterUpload()
230      */
231     public void afterUpload() {
232         if (this.uploadInputStream != null) {
233             try {
234                 this.uploadInputStream.close();
235             } catch (IOException ioe) {
236                 // Let's ignore it.
237             }
238             this.uploadInputStream = null;
239         }
240         // Transmission to the 'real' FileData
241         this.fileData.afterUpload();
242     }
243 
244     /**
245      * Clean any local resources, to allow retrying to upload this file. The
246      * file remains prepared..
247      */
248     public void beforeRetry() {
249         // Reset of our upload counter.
250         this.uploadRemainingLength = this.fileData.getUploadLength();
251 
252         if (this.uploadInputStream != null) {
253             try {
254                 this.uploadInputStream.close();
255             } catch (IOException ioe) {
256                 // Let's ignore it.
257                 this.uploadPolicy.displayDebug("[Warning] Ignoring " + ioe.getClass().getName()
258                         +" in UploadFileData.beforeRetry() ["
259                         +ioe.getMessage()+"]",
260                         30);
261             }
262             this.uploadInputStream = null;
263         }
264     }
