81  /**
82   * This class implements all {@link wjhk.jupload2.policies.UploadPolicy}
83   * methods. Its way of working is he same as the JUpload version 1. <BR>
84   * The simplest way to use this policy is given in the presentation of
85   * {@link UploadPolicy}. The DefaultUploadPolicy is used when no
86   * <I>uploadPolicy</I> parameter is given to the juploadContext, or this
87   * parameter has 'DefaultUploadPolicy' as a value. <BR>
88   * <P>
89   * The <U>default behavior</U> is representated below. It can be overridden by
90   * adding parameters to the juploadContext. All available parameters are shown
91   * in the presentation of {@link UploadPolicy}.
92   * </P>
93   * <UL>
94   * <LI>Default implementation for all
95   * {@link wjhk.jupload2.policies.UploadPolicy} methods.
96   * <LI>Files are uploaded all in one HTTP request.
97   * <LI>No handling for particular kind of files: files are transmitted without
98   * any transformation.
99   * <LI>The file are transmitted to the server with the navigator cookies,
100  * userAgent and Protocol (see also the readCookieFromNavigator and
101  * serverProtocol juploadContext parameter). This make upload occurs within the
102  * current user session on the server. So, it allows right management and
103  * context during the management of uploaded files, on the server.
104  * </UL>
105  * 
106  * @author etienne_sf
107  * @version $Revision: 1581 $
108  */

551 	/** @see UploadPolicy#start() */
552 	public void start() {
553 		// The current context may add any specific headers.
554 		if (getReadCookieFromNavigator()) {
555 			this.juploadContext.readCookieFromNavigator(this.headers);
556 		}
557 		if (getReadUserAgentFromNavigator()) {
558 			this.juploadContext.readUserAgentFromNavigator(this.headers);
559 		}
560 		
561         // Let's touch the server, to test that everything is Ok. Take care,
562         // this is the only place where we override the default value, by null:
563         // the default value will be used by the HttpConnect.getProtocol()
564         // method.
565         // Also, in FTP mode, there can be no default value.
566 		//
567 		// Must be done AFTER reading the cookie and the userAgent.
568 		//
569         HttpProtocolFinderThread.computeServerProtocol(this, juploadContext
570                 .getParameter(PROP_SERVER_PROTOCOL, null));
571 	}

588 	 * True, if the stringUploadSuccess is null or empty (no test at all). <LI>
589 	 * True, if the stringUploadSuccess string is present in the
590 	 * serverOutputBody. <LI>True, If previous condition is not filled, but the
591 	 * HTTP header "HTTP(.*)200OK$" is present: the test is currently non
592 	 * blocking, because I can not test all possible HTTP configurations.<BR>
593 	 * <LI>False if the previous conditions are not fullfilled. </DIR> <BR>
594 	 * This method also looks for the stringUploadWarning regular expression.
595 	 * Each time it is matched, the found message is displayed to the user.
596 	 * 
597 	 * @param status
598 	 *            The HTTP response code
599 	 * @param msg
600 	 *            The status message from the first line of the response (e.g.
601 	 *            "200 OK").
602 	 * @param body
603 	 *            The body of the HTTP answer.
604 	 * @return True or False, indicating if the upload is a success or not.
605 	 * @see UploadPolicy#checkUploadSuccess(int, String, String)
606 	 */
607 	public boolean checkUploadSuccess(int status, String msg, String body)
608 			throws JUploadException {
609 		boolean bReturn = false;
610 
611 		if (getDebugLevel() > 100) {
612 			// Let's have a little time to check the upload messages written on
613 			// the progress bar.
614 			try {
615 				Thread.sleep(300);
616 			} catch (InterruptedException e) {
617 			}
618 		}
619 
620 		this.lastResponseBody = body;
621 		this.lastResponseMessage = msg;
622 		displayDebug("HTTP status: " + msg, 30);
623 		// HTTP-100 correction, thanks to Marc Reidy
624 		if ((status != 200) && (status != 100))
625 			throw new JUploadExceptionUploadFailed("Received HTTP status "
626 					+ msg);
627 
628                 // HTTP-100 "continue", in case we're uploading
629                 // to an ASP.NET development server. We should
630                 // continue sending...
631                 if(status == 100)
632                     return true;
633 
634 		// Let's analyze the body returned, line by line. The end of line
635 		// character may be CR, LF, or CRLF. We navigate through the body, and
636 		// replace any end of line character by a uniform CRLF.
637 		Matcher matcherError, matcherWarning;
638 		String line;
639 		Pattern p = Pattern.compile("[\\r\\n]", Pattern.MULTILINE);
640 		String[] lines = p.split(body);
641 		StringBuffer sbBodyWithUniformCRLF = new StringBuffer(body.length());
642 		for (int i = 0; i < lines.length; i += 1) {
643 			line = lines[i];
644 			sbBodyWithUniformCRLF.append(line).append("\r\n");
645 
646 			// FIXME some empty lines are given by the server
647 			// Let's remove the empty line: with the p pattern, a multiline is
648 			// generated each time a \r\n is received, that is: for each line.
649 			if (line == null || line.equals("")) {
650 				// An empty line. Let's go the next line.
651 				continue;
652 			}
653 
654 			// Check if this is a success
655 			// The success string should be in the http body
656 			if (getStringUploadSuccess() != null
657 					&& !getStringUploadSuccess().equals("")) {
658 				if (this.patternSuccess.matcher(line).matches()) {
659 					// We go on. There may be some WARNING message, hereafter.
660 					bReturn = true;
661 				}
662 			}
663 
664 			// Check if this is an error
665 			if (getStringUploadError() != null
666 					&& !getStringUploadError().equals("")) {
667 				matcherError = this.patternError.matcher(line);
668 				if (matcherError.matches()) {
669 					String errmsg = "An error occurs during upload (but the applet couldn't find the error message)";
670 					if (matcherError.groupCount() > 0) {
671 						if (!matcherError.group(1).equals("")) {
672 							// Let's do a (very simple) formatting: one line to
673 							// 100 characters
674 							errmsg = formatMessage(matcherError.group(1));
675 						}
676 					}
677 					this.lastResponseMessage = errmsg;
678 					throw new JUploadExceptionUploadFailed(errmsg);
679 				}
680 			}// getStringUploadError
681 
682 			// Check if this is an warning
683 			if (getStringUploadWarning() != null
684 					&& !getStringUploadWarning().equals("")) {
685 				matcherWarning = this.patternWarning.matcher(line);
686 				if (matcherWarning.matches()) {
687 					String warnmsg = "A warning occurs during upload (but the applet couldn't find the warning message)";
688 					if (matcherWarning.groupCount() > 0) {
689 						if (!matcherWarning.group(1).equals("")) {
690 							warnmsg = formatMessage(matcherWarning.group(1));
691 						}
692 					}
693 					this.lastResponseMessage = warnmsg;
694 					displayWarn(warnmsg);
695 					alertStr(warnmsg);
696 				}
697 			}// getStringUploadWarning
698 
699 		}// while(st.hasMoreTokens())
700 
701 		if (bReturn) {
702 			return true;
703 		}
704 
705 		// We found no stringUploadSuccess nor stringUploadError
706 		if (getStringUploadSuccess() == null
707 				|| getStringUploadSuccess().equals("")) {
708 			// No chance to check the correctness of this upload. -> Assume Ok
709 			return true;
710 		}
711 
712 		// stringUploadSuccess was defined but we did not find it.
713 		// This is most certainly an error as http-status 200 does *not* refer
714 		// to the correctness of the content. It merely means that the protocol
715 		// handling was ok. -> throw an exception
716 		throw new JUploadExceptionUploadFailedSuccessNotFound(getClass()
717 				.getName()
718 				+ ".checkUploadSuccess(): The regexp string \""
719 				+ getStringUploadSuccess()
720 				+ "\" was not found in the response body");
721 	} // checkUploadSuccess

723 	/**
724 	 * Generation of a formatted error message, so that it can be displayed in a
725 	 * DialogBox. That is: each line that contains more than
726 	 * {@link #DIALOG_MESSAGE_MAX_LINE_LENGTH} is truncated to
727 	 * DIALOG_MESSAGE_MAX_LINE_LENGTH.
728 	 * 
729 	 * @param msg
730 	 * @return
731 	 */
732 	protected String formatMessage(String msg) {
733 		StringBuffer sbMsg = new StringBuffer();
734 		String[] lines = msg.split("\\\\n");
735 		for (int i = 0; i < lines.length; i += 1) {
736 			// For each line, we truncate it, if it's too long.
737 			String line = lines[i];
738 			for (int j = 0, remaining = line.length(); remaining > 0; remaining -= DIALOG_MESSAGE_MAX_LINE_LENGTH, j += 1) {
739 				if (remaining <= DIALOG_MESSAGE_MAX_LINE_LENGTH) {
740 					// It's the last loop.
741 					sbMsg.append(line.substring(j
742 							* DIALOG_MESSAGE_MAX_LINE_LENGTH));
743 				} else {
744 					sbMsg.append(line.substring(j
745 							* DIALOG_MESSAGE_MAX_LINE_LENGTH, (j + 1)
746 							* DIALOG_MESSAGE_MAX_LINE_LENGTH));
747 					sbMsg.append("\n");
748 				}
749 			}
750 			// End we add the EOL characters, for all lines, but the last one.
751 			if (i < lines.length - 1) {
752 				sbMsg.append("\n");
753 			}
754 		}
755 
756 		return sbMsg.toString();
757 	}

980 	 * 
981 	 * @param errorText
982 	 * @param exception
983 	 */
984 	private void logErr(String errorText, Exception exception) {
985 		if (exception == null) {
986 			setLastException(new JUploadException("errorText"));
987 		} else if (exception instanceof JUploadException) {
988 			setLastException((JUploadException) exception);
989 		} else {
990 			setLastException(new JUploadException(exception));
991 		}
992 
993 		// Default behavior: if debugLevel is 0, and an error occurs, we force
994 		// the debug level to 1: this makes the log window become visible, if it
995 		// was hidden.
996 		if (getDebugLevel() == 0)
997 			setDebugLevel(1);
998 
999 		String exceptionMsg = null;
1000 		String exceptionClassName = null;
1001 		String logMsg = errorText;
1002 
1003 		// First, we construct the exception class name.
1004 		if (exception == null) {
1005 			exceptionClassName = "";
1006 		} else if (exception instanceof JUploadException) {
1007 			exceptionClassName = "["
1008 					+ ((JUploadException) exception).getClassNameAndClause()
1009 					+ "] ";
1010 		} else {
1011 			exceptionClassName = "[" + exception.getClass().getName() + "] ";
1012 		}
1013 
1014 		// Then, the message body can be completed by the exception message.
1015 		if (exception != null) {
1016 			// Ok, we have an exception.
1017 			if (exception.getCause() != null) {
1018 				exceptionMsg = exception.getCause().getMessage();
1019 			} else {
1020 				exceptionMsg = exception.getMessage();
1021 			}
1022 			logMsg = exceptionMsg + " (" + errorText + ")";
1023 		}
1024 
1025 		// Add the message to the log window
1026 		displayMsg("[ERROR]", exceptionClassName + logMsg);
1027 		// Let's display the stack trace, if relevant.
1028 		displayStackTrace(exception);
1029 
1030 		// Then we copy the debug output to the clipboard, and say it to the
1031 		// current user.
1032 		if (this.juploadContext.getUploadPanel() != null
1033 				&& getDebugLevel() >= 99) {
1034 			// Ok, the applet has been fully built.
1035 			this.juploadContext.getUploadPanel().copyLogWindow();
1036 			alert("messageLogWindowCopiedToClipboard");

1042 	 * If debug is off, the log window may not be visible. We switch the debug
1043 	 * to on, to be sure that some information will be displayed to the user. <BR>

1050 		// Then, we display it to the user.
1051 		String alertMsg = errorText;
1052 		// Then, the message body can be completed by the exception message.
1053 		if (exception != null && (errorText == null || errorText.equals(""))) {
1054 			// Ok, we have an exception.
1055 			if (exception.getCause() != null) {
1056 				alertMsg = exception.getCause().getMessage();
1057 			} else {
1058 				alertMsg = exception.getMessage();
1059 			}
1060 		}
1061 
1062 		// The message displayed depend on the debug level:
1063 		if (getDebugLevel() >= 30 && exception != null) {
1064 			alertMsg = exception.getClass().getName() + ": " + alertMsg;
1065 		}
1066 
1067 		// Display the message to the user. The kind of alert box depends on the
1068 		// given options:
1069 		int buttonClicked = 0;
1070 		switch (optionTypes) {
1071 		case -1:
1072 			// Standard message box.
1073 			alertStr(alertMsg);

1076 		case JOptionPane.OK_CANCEL_OPTION:

1079 			buttonClicked = confirmDialogStr(alertMsg, optionTypes);

1081 		default:

1084 			alertStr(alertMsg);

1088 			alertStr(msg);
1089 			logErr(msg, null);
1090 		}
1091 
1092 		// First, we log the error.
1093 		logErr(errorText, exception);
1094
