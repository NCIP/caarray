50  /**
51   * This class contains all data about files to upload as a picture. It adds the
52   * following elements to the {@link wjhk.jupload2.filedata.FileData} class :<BR>
53   * <UL>
54   * <LI>Ability to define a target format (to convert pictures to JPG before
55   * upload, for instance)
56   * <LI>Optional definition of a maximal width and/or height.
57   * <LI>Ability to rotate a picture, with {@link #addRotation(int)}
58   * <LI>Ability to store a picture into a BufferedImage. This is actualy a bad
59   * idea within an applet (should run within a java application) : the applet
60   * runs very quickly out of memory. With pictures from my Canon EOS20D (3,5M), I
61   * can only display two pictures. The third one generates an out of memory
62   * error, despite the System.finalize and System.gc I've put everywhere in the
63   * code!
64   * </UL>
65   * 
66   * @author etienne_sf
67   * @version $Revision: 1541 $

175     /**
176      * Free any available memory. This method is called very often here, to be
177      * sure that we don't use too much memory. But we still run out of memory in
178      * some case.
179      * 
180      * @param caller Indicate the method or treatment from which this method is
181      *            called.
182      * @param uploadPolicy The current upload policy is not available, to this
183      *            static method...
184      */
185     public static void freeMemory(String caller, UploadPolicy uploadPolicy) {
186         Runtime rt = Runtime.getRuntime();
187 
188         rt.runFinalization();
189         rt.gc();
190 
191         if (uploadPolicy.getDebugLevel() >= 50) {
192             uploadPolicy.displayDebug("freeMemory (after " + caller + ") : "
193                     + rt.freeMemory() + " (maxMemory: " + rt.maxMemory()
194                     + ", totalMemory: " + rt.totalMemory() + ")", 50);
195         }
196     }

198     /**
199      * If this pictures needs transformation, a temporary file is created. This
200      * can occurs if the original picture is bigger than the maxWidth or
201      * maxHeight, of if it has to be rotated. This temporary file contains the
202      * transformed picture. <BR>
203      * The call to this method is optional, if the caller calls
204      * {@link #getUploadLength()}. This method calls beforeUpload() if the
205      * uploadLength is unknown.
206      */
207     @Override
208     public void beforeUpload() throws JUploadException {
209         uploadPolicy.displayDebug(this.hashCode()
210                 + "|Entering PictureFileData.beforeUpload()", 95);
211 
212         if (!this.preparedForUpload) {
213             if (this.uploadLength < 0) {
214                 try {
215                     // Picture management is a big work. Let's try to free some
216                     // memory.
217                     freeMemory(
218                             "Picture manabeforeUpload(): before initTransformedPictureFile",
219                             uploadPolicy);
220 
221                     // Get the transformed picture file, if needed.
222                     initTransformedPictureFile();
223 
224                     // More debug output, to understand where the applet
225                     // freezes.
226                     uploadPolicy
227                             .displayDebug(
228                                     this.getClass().getName()
229                                             + ".beforeUpload(): after call to initTransformedPictureFile()",
230                                     100);
231 
232                 } catch (OutOfMemoryError e) {
233                     // Oups ! My EOS 20D has too big pictures to handle more
234                     // than
235                     // two pictures in a navigator applet !!!!!
236                     // :-(
237                     //
238                     // We don't transform it. We clean the file, if it has been
239                     // created.
240                     // More debug output, to understand where the applet
241                     // freezes.
242                     uploadPolicy.displayDebug(this.getClass().getName()
243                             + ".beforeUpload(): OutOfMemoryError", 30);
244                     deleteTransformedPictureFile();
245                     deleteWorkingCopyPictureFile();
246 
247                     // Let's try to free some memory.
248                     freeMemory("beforeUpload(): in OutOfMemoryError",
249                             uploadPolicy);
250 
251                     tooBigPicture();
252                 }
253 
254                 // If the transformed picture is correctly created, we'll upload
255                 // it.
256                 // Else we upload the original file.
257                 synchronized (this) {
258                     if (this.transformedPictureFile != null) {
259                         this.uploadLength = this.transformedPictureFile
260                                 .length();
261                         setMimeTypeByExtension(this.targetPictureFormat);
262                     } else {
263                         this.uploadLength = getFile().length();
264                     }
265                 }
266             }
267         }
268 
269         // Let's check that everything is Ok
270         // More debug output, to understand where the applet freezes.
271         uploadPolicy.displayDebug(this.getClass().getName()
272                 + ".beforeUpload(): before call to super.beforeUpload()", 100);
273 
274         super.beforeUpload();
275 
276         // Let's check that everything is Ok
277         // More debug output, to understand where the applet freezes.
278         uploadPolicy.displayDebug(this.getClass().getName()
279                 + ".beforeUpload(): after call to super.beforeUpload()", 100);
280     }

300     /**
301      * This function create an input stream for this file. The caller is
302      * responsible for closing this input stream. <BR>
303      * This function assumes that the {@link #getUploadLength()} method has
304      * already be called : it is responsible for creating the temporary file (if
305      * needed). If not called, the original file will be sent.
306      * 
307      * @return An inputStream
308      */
309     @Override
310     public synchronized InputStream getInputStream() throws JUploadException {
311         uploadPolicy.displayDebug(this.hashCode()
312                 + "|Entering PictureFileData.getInputStream()", 95);
313         if (!this.preparedForUpload) {
314             throw new IllegalStateException("The file " + getFileName()
315                     + " is not prepared for upload");
316         }
317         // Do we have to transform the picture ?
318         if (this.transformedPictureFile != null) {
319             try {
320                 return new FileInputStream(this.transformedPictureFile);
321             } catch (FileNotFoundException e) {
322                 throw new JUploadIOException(e);
323             }
324         }
325         // Otherwise : we read the file, in the standard way.
326         return super.getInputStream();
327     }

344     /**
345      * This method creates a new Image, from the current picture. The resulting
346      * width and height will be less or equal than the given maximum width and
347      * height. The scale is maintained. Thus the width or height may be inferior
348      * than the given values.
349      * 
350      * @param canvas The canvas on which the picture will be displayed.
351      * @param shadow True if the pictureFileData should store this picture.
352      *            False if the pictureFileData instance should not store this
353      *            picture. Store this picture avoid calculating the image each
354      *            time the user selects it in the file panel.
355      * @return The rescaled image.
356      * @throws JUploadException Encapsulation of the Exception, if any would
357      *             occurs.
358      */
359     public Image getImage(Canvas canvas, boolean shadow)
360             throws JUploadException {
361         Image localImage = null;
362 
363         // ////////////////////////////////////////////////////////////////////////
364         // ////////////// Some preliminary tests.
365         // ////////////////////////////////////////////////////////////////////////
366 
367         if (canvas == null) {
368             throw new JUploadException(
369                     "canvas null in PictureFileData.getImage");
370         }
371 
372         if (shadow && this.offscreenImage != null) {
373             return this.offscreenImage;
374         }
375 
376         int canvasWidth = canvas.getWidth();
377         int canvasHeight = canvas.getHeight();
378         if (canvasWidth <= 0 || canvasHeight <= 0) {
379             uploadPolicy
380                     .displayDebug(
381                             "canvas width and/or height null in PictureFileData.getImage()",
382                             1);
383             return null;
384         }
385 
386         if (!this.isPicture) {
387             uploadPolicy
388                     .displayWarn("canvas width and/or height null in PictureFileData.getImage(). PictureFileData.getImage will return null");
389             return null;
390         }
391         // ////////////////////////////////////////////////////////////////////////
392         // ////////////// End of preliminary tests: let's work.
393         // ////////////////////////////////////////////////////////////////////////
394 
395         try {
396             // First: load the picture.
397             ImageReaderWriterHelper irwh = new ImageReaderWriterHelper(
398                     (PictureUploadPolicy) uploadPolicy, this);
399             BufferedImage sourceImage = irwh.readImage(0);
400             irwh.dispose();
401             irwh = null;
402             ImageHelper ih = new ImageHelper(
403                     (PictureUploadPolicy) uploadPolicy, this, canvasWidth,
404                     canvasHeight, this.quarterRotation);
405             localImage = ih.getBufferedImage(
406                     ((PictureUploadPolicy) uploadPolicy)
407                             .getHighQualityPreview(), sourceImage);
408             // We free memory ASAP.
409             sourceImage.flush();
410             sourceImage = null;
411         } catch (OutOfMemoryError e) {
412             // Too bad
413             localImage = null;
414             tooBigPicture();
415         }
416 
417         // We store it, if asked to.
418         if (shadow) {
419             this.offscreenImage = localImage;
420         }
421 
422         freeMemory("end of " + this.getClass().getName() + ".getImage()",
423                 uploadPolicy);
424 
425         // The picture is now loaded. We clear the progressBar
426         uploadPolicy.getContext().getUploadPanel().getPreparationProgressBar()
427                 .setValue(0);
428 
429         return localImage;
430     }// getImage

432     /**
433      * This function is used to rotate the picture. The current rotation state
434      * is kept in the quarterRotation private attribute.
435      * 
436      * @param quarter Number of quarters (90 degrees) the picture should rotate.
437      *            1 means rotating of 90 degrees clockwise. Can be negative.
438      */
439     public void addRotation(int quarter) {
440         this.quarterRotation += quarter;
441 
442         // We'll have to recalculate the upload length, as the resulting file is
443         // different.
444         // If any file has been prepared, they must be deleted
445         deleteWorkingCopyPictureFile();
446         deleteTransformedPictureFile();
447         this.uploadLength = -1;
448 
449         // We keep the 'quarter' in the segment [0;4[
450         while (this.quarterRotation < 0) {
451             this.quarterRotation += 4;
452         }
453         while (this.quarterRotation >= 4) {
454             this.quarterRotation -= 4;
455         }
456 
457         // We need to change the precalculated picture, if any
458         if (this.offscreenImage != null) {
459             this.offscreenImage.flush();
460             this.offscreenImage = null;
461         }
462     }

484      * File.deleteOnExit() is pretty unreliable, especially in applets.
485      * Therefore the applet provides a callback which is executed during applet
486      * termination. This method performs the actual cleanup.
487      */
488     public synchronized void deleteTransformedPictureFile() {
489         uploadPolicy.displayDebug(this.hashCode()
490                 + "|Entering PictureFileData.deleteTransformedPictureFile()",
491                 95);
492         // Free the temporary file ... if any.
493         if (null != this.transformedPictureFile
494                 && uploadPolicy.getDebugLevel() <= 100) {
495             if (!this.transformedPictureFile.delete()) {
496                 uploadPolicy.displayWarn("Unable to delete "
497                         + this.transformedPictureFile.getName());
498             }
499             this.transformedPictureFile = null;
500 
501         }
502     }

504     /**
505      * Creation of a temporary file, that contains the transformed picture. For
506      * instance, it can be resized or rotated. This method doesn't throw
507      * exception when there is an IOException within its procedure. If an
508      * exception occurs while building the temporary file, the exception is
509      * caught, a warning is displayed, the temporary file is deleted (if it was
510      * created), and the upload will go on with the original file. <BR>
511      * Note: any JUploadException thrown by a method called within
512      * getTransformedPictureFile() will be thrown within this method.
513      */
514     void initTransformedPictureFile() throws JUploadException {
515         uploadPolicy.displayDebug(this.hashCode()
516                 + "|Entering PictureFileData.initTransformedPictureFile()", 95);
517         int targetMaxWidth;
518         int targetMaxHeight;
519 
520         // If the image is rotated, we compare to realMaxWidth and
521         // realMaxHeight, instead of maxWidth and maxHeight. This allows
522         // to have a different picture size for rotated and not rotated
523         // pictures. See the UploadPolicy javadoc for details ... and a
524         // good reason ! ;-)
525         if (this.quarterRotation == 0) {
526             targetMaxWidth = ((PictureUploadPolicy) uploadPolicy).getMaxWidth();
527             targetMaxHeight = ((PictureUploadPolicy) uploadPolicy)
528                     .getMaxHeight();
529         } else {
530             targetMaxWidth = ((PictureUploadPolicy) uploadPolicy)
531                     .getRealMaxWidth();
532             targetMaxHeight = ((PictureUploadPolicy) uploadPolicy)
533                     .getRealMaxHeight();
534         }
535 
536         // Some Helper will .. help us !
537         // I like useful comment :-)
538         ImageHelper imageHelper = new ImageHelper(
539                 (PictureUploadPolicy) uploadPolicy, this, targetMaxWidth,
540                 targetMaxHeight, this.quarterRotation);
541 
542         // Should transform the file, and do we already created the transformed
543         // file ?
544         synchronized (this) {
545             if (imageHelper.hasToTransformPicture()
546                     && this.transformedPictureFile == null) {
547 
548                 // We have to create a resized or rotated picture file, and all
549                 // needed information.
550                 // ...let's do it
551                 try {
552                     createTranformedPictureFile(imageHelper);
553                 } catch (JUploadException e) {
554                     // Hum, too bad.
555                     // if any file was created, we remove it.
556                     deleteTransformedPictureFile();
557                     throw e;
558                 }
559             }
560         }
561     }// end of initTransformedPictureFile

563     /**
564      * Creates a transformed picture file of the given max width and max height.
565      * If the {@link #transformedPictureFile} attribute is not set before
566      * calling this method, it will be set. If set before, the existing
567      * {@link #transformedPictureFile} is replaced by the newly transformed
568      * picture file. It is cleared if an error occured. <BR>
569      * 
570      * @param imageHelper The {@link ImageHelper} that was initialized with
571      *            current parameters.
572      */
573     synchronized void createTranformedPictureFile(ImageHelper imageHelper)
574             throws JUploadException {
575         uploadPolicy.displayDebug(this.hashCode()
576                 + "|Entering PictureFileData.createTransformedPictureFile()",
577                 95);
578 
579         IIOMetadata metadata = null;
580         IIOImage iioImage = null;
581         BufferedImage originalImage = null;
582         BufferedImage transformedImage = null;
583         ImageReaderWriterHelper imageWriterHelper = new ImageReaderWriterHelper(
584                 (PictureUploadPolicy) uploadPolicy, this);
585         boolean transmitMetadata = ((PictureUploadPolicy) uploadPolicy)
586                 .getPictureTransmitMetadata();
587 
588         // Creation of the transformed picture file.
589         createTransformedTempFile();
590         this.targetPictureFormat = imageWriterHelper.getTargetPictureFormat();
591         imageWriterHelper.setOutput(this.transformedPictureFile);
592 
593         // How many picture should we read from the input file.
594         // Default number of pictures is one.
595         int nbPictures = 1;
596         // For gif file, we put a max to MAX_VALUE, and we check the
597         // IndexOutOfBoundsException to identify when we've read all pictures
598         if (getExtension(getFile()).equalsIgnoreCase("gif")) {
599             nbPictures = Integer.MAX_VALUE;
600         }
601         uploadPolicy.displayDebug(
602                 "Reading image with imageWriterHelper.readImage(i)", 50);
603         // Now, we have to read each picture from the original file, apply
604         // the calculated transformation, and write each transformed picture
605         // to the writer.
606         // As indicated in javadoc for ImageReader.getNumImages(), we go
607         // through pictures, until we get an IndexOutOfBoundsException.
608         try {
609             for (int i = 0; i < nbPictures; i += 1) {
610                 originalImage = imageWriterHelper.readImage(i);
611                 transformedImage = imageHelper.getBufferedImage(true,
612                         originalImage);
613 
614                 // If necessary, we load the metadata for the current
615                 // picture
616                 if (transmitMetadata) {
617                     metadata = imageWriterHelper.getImageMetadata(i);
618                 }
619 
620                 iioImage = new IIOImage(transformedImage, null, metadata);
621                 imageWriterHelper.write(iioImage);
622 
623                 // Let's clear picture, to force getBufferedImage to read a new
624                 // one,
625                 // in the next loop.
626                 if (originalImage != null) {
627                     originalImage.flush();
628                     originalImage = null;
629                 }
630             }// for
631         } catch (IndexOutOfBoundsException e) {
632             // Was sent by imageWriterHelper.readImage(i)
633             // Ok, no more picture to read. We just want to go out of
634             // the loop. No error.
635             uploadPolicy.displayDebug(
636                     "IndexOutOfBoundsException catched: end of reading for file "
637                             + getFileName(), 10);
638         }
639 
640         if (originalImage != null) {
641             originalImage.flush();
642             originalImage = null;
643         }
644 
645         // Let's free any used resource.
646         imageWriterHelper.dispose();
647 
648     }

682     synchronized void createTransformedTempFile() throws JUploadIOException {
683         uploadPolicy.displayDebug(this.hashCode()
684                 + "|Entering PictureFileData.createTransformedTempFile()", 95);

690             } catch (IOException e) {

695                     "deleteTransformedPictureFile");
696             uploadPolicy.displayDebug("Using transformed temp file "
697                     + this.transformedPictureFile.getAbsolutePath() + " for "
698                     + getFileName(), 30);
699         }

706      * @throws JUploadIOException
707      * @see #getOriginalHeight()
708      * @see #getOriginalWidth()
709      */
710     void initWidthAndHeight() throws JUploadIOException {
711         // Is it a picture?
712         if (this.isPicture
713                 && (this.originalHeight < 0 || this.originalWidth < 0)) {
714             // Ok: it's a picture and is original width and height have not been
715             // loaded yet.
716             // In the windows world, file extension may be in upper case, which
717             // is not compatible with the core Java API.
718             Iterator<ImageReader> iter = ImageIO
719                     .getImageReadersByFormatName(getFileExtension()
720                             .toLowerCase());
721             if (iter.hasNext()) {
722                 // It's a picture: we store its original width and height, for
723                 // further calculation (rescaling and rotation).
724                 try {
725                     FileImageInputStream fiis = new FileImageInputStream(
726                             getFile());
727                     ImageReader ir = iter.next();
728                     ir.setInput(fiis);
729                     this.originalHeight = ir.getHeight(0);
730                     this.originalWidth = ir.getWidth(0);
731                     ir.dispose();
732                     fiis.close();
733                 } catch (IOException e) {
734                     throw new JUploadIOException("PictureFileData()", e);
735                 }
736             }
737         }

863     /**
864      * Returns an ImageIcon for the given file, resized according to the given
865      * dimensions. If the original file contains a pictures smaller than these
866      * width and height, the picture is returned as is (nor resized).
867      * 
868      * @param pictureFile The file, containing a picture, from which the user
869      *            wants to extract a static picture.
870      * @param maxWidth The maximum allowed width for the static picture to
871      *            generate.
872      * @param maxHeight The maximum allowed height for the static picture to
873      *            generate.
874      * @param uploadPolicy The current upload policy, for logging when calling
875      *            freeMemory.
876      * @return The created static picture, or null if the file is null.
877      * @throws JUploadException If the ImageIcon can not be loaded.
878      */
879     public static ImageIcon getImageIcon(File pictureFile, int maxWidth,
880             int maxHeight, UploadPolicy uploadPolicy) throws JUploadException {
881         ImageIcon imageIcon = null;
882 
883         if (pictureFile != null) {
884             BufferedImage tmpImage = null;
885             try {
886                 // We need a reader for this picture.
887                 if (ImageIO.getImageReaders(
888                         new FileImageInputStream(pictureFile)).hasNext()) {
889                     tmpImage = ImageIO.read(pictureFile);
890                 }
891             } catch (IOException e) {
892                 throw new JUploadIOException(
893                         "An error occured while loading the image for "
894                                 + pictureFile.getAbsolutePath(), e);
895             }
896             if (tmpImage != null) {
897                 // Let's calculate the asked icon.
898                 double scaleWidth = ((double) maxWidth) / tmpImage.getWidth();
899                 double scaleHeight = ((double) maxHeight)
900                         / tmpImage.getHeight();
901                 double scale = Math.min(scaleWidth, scaleHeight);
902 
903                 if (scale < 1) {
904                     int width = (int) (scale * tmpImage.getWidth());
905                     int height = (int) (scale * tmpImage.getHeight());
906                     // We need to miniaturize the current Icon
907                     Image rescaledImage = tmpImage.getScaledInstance(width,
908                             height, Image.SCALE_FAST);
909                     BufferedImage tempBufferedImage = new BufferedImage(width,
910                             height, BufferedImage.TYPE_INT_BGR);
911                     tempBufferedImage.getGraphics().drawImage(rescaledImage, 0,
912                             0, null);
913                     imageIcon = new ImageIcon(tempBufferedImage);
914                     uploadPolicy.displayDebug("Icon generated (witdh="
915                             + imageIcon.getIconWidth() + ", imageheight="
916                             + imageIcon.getIconHeight() + ")", 80);
917                 }
918                 tmpImage.flush();
919                 tmpImage = null;
920 
921                 freeMemory("PictureFileData.getImageIcon()", uploadPolicy);
922             }
923         }
924         return imageIcon;
925     }
