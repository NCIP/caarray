37  /**
38   * This class is based on the {@link FileUploadThread} class. It's an abstract
39   * class that contains the default implementation for the
40   * {@link FileUploadThread} interface. <BR>
41   * It contains the following abstract methods, which must be implemented in the
42   * children classes. These methods are called in this order: <DIR> <LI>For each
43   * upload request (for instance, upload of 3 files with nbFilesPerRequest to 2,
44   * makes 2 request: 2 files, then the last one): <DIR> <LI><I>try</I> <LI>
45   * {@link #startRequest}: start of the UploadRequest. <LI>Then, for each file to
46   * upload (according to the nbFilesPerRequest and maxChunkSize applet
47   * parameters) <DIR> <LI>beforeFile(int) is called before writting the bytes for
48   * this file (or this chunk) <LI>afterFile(int) is called after writting the
49   * bytes for this file (or this chunk) </DIR> <LI>finishRequest() </DIR></LI>
50   * <I>finally</I>cleanRequest() <LI>Call of cleanAll(), to clean up any used
51   * resources, common to the whole upload. </DIR>
52   */
53  public abstract class DefaultFileUploadThread extends Thread implements
54          FileUploadThread {

299     /**
300      * This method waits for a packet on the packetQueue. Then, it calls the
301      * doUpload() method, to send these files to the server.
302      */
303     @Override
304     final public void run() {
305         this.uploadPolicy.displayDebug("Start of the FileUploadThread", 5);
306 
307         // We'll stop the upload if an error occurs. So the try/catch is
308         // outside the while.
309         while (!this.fileUploadManagerThread.isUploadFinished()) {
310             UploadFilePacket packet = null;
311 
312             try {
313                 // We take the next packet. This method will block until a
314                 // packet is ready.
315                 packet = packetQueue.take();
316 
317                 // If the packet is 'poisonned', then it's the standard end of
318                 // work.
319                 if (packet.isPoisonned()) {
320                     break;
321                 }
322 
323                 // /////////////////////////////////////////////////////////////////////////////////
324                 // Let's go to work : THIS IS THE UPLOAD, surrounded by the
325                 // RESUME LOOP
326                 // /////////////////////////////////////////////////////////////////////////////////
327                 nbRetry = 0;
328                 while (true) {
329                     try {
330                         // Let's try to upload the current packet.
331                         doUpload(packet);
332 
333                         // If we are here, the last upload is a success. Let's
334                         // exit the loop.
335                         break;
336                     } catch (JUploadException jue) {
337                         // manageRetry throw the exception, if no retry should
338                         // be done.
339                         manageRetry(jue);
340 
341                         // If we are here, the applet should retry the upload.
342                         // We let it loop again.
343                         nbRetry += 1;
344 
345                         // We must clean the resources of the previous attempt.
346                         beforeRetry(packet);
347                     }
348                 }// while(resume)
349                 // /////////////////////////////////////////////////////////////////////////////////
350                 // //////////////// ENF OF RESUME LOOP
351                 // /////////////////////////////////////////////////////////////////////////////////
352 
353                 this.uploadPolicy.displayDebug("After do upload", 50);
354             } catch (InterruptedException e) {
355                 this.uploadPolicy.displayWarn(this.getClass().getName()
356                         + ".run(): received in " + e.getClass().getName()
357                         + ", exiting...");
358                 break;
359             } catch (JUploadException e) {
360                 if (this.fileUploadManagerThread.isUploadFinished()) {
361                     // We ignore the error
362                     this.uploadPolicy.displayWarn("Ignoring "
363                             + e.getClass().getName()
364                             + " because upload is finished");
365                 } else {
366                     this.fileUploadManagerThread.setUploadException(e);
367                 }
368             } catch (JUploadInterrupted e) {
369                 // The upload has been interrupted, probably by the user
370                 // (stop
371                 // button). The fileManagerUploadThread aleady knows this.
372                 this.uploadPolicy.displayInfo("Upload stopped by the user");
373                 this.uploadPolicy.displayDebug(e.getMessage(), 30);
374 			} finally {
375 				// Let's free any locked resource for the current packet.
376 				// This is done here, to allow the resume feature (and, even in
377 				// case an error occurs, we free resources only after the last
378 				// retry)
379 				for (UploadFileData uploadFileData : packet) {
380 					if (uploadFileData.isPreparedForUpload()) {
381 						uploadFileData.afterUpload();
382 					}
383 				}
384             }
385         }// while (!isUploadFinished)
386 
387         this.uploadPolicy.displayDebug("End of the FileUploadThread", 5);
388     }// run

390     /**
391      * @param jue
392      * @throws JUploadException
393      */
394     private void manageRetry(JUploadException jue) throws JUploadException {
395         String exceptionCauseClass = (jue.getCause() == null) ? "no exception cause"
396                 : jue.getCause().getClass().getName();
397         String errMsg = jue.getClass().getName() + " (" + jue.getMessage()
398                 + "), caused by: " + exceptionCauseClass;
399 
400         if (this.fileUploadManagerThread.isUploadFinished()) {
401             // The upload is stopped. This error may be caused
402             this.uploadPolicy
403                     .displayWarn("The following error occurs, but the upload is stopped, ignoring it ]"
404                             + errMsg + "]");
405             throw jue;
406         } else if (jue.getCause() instanceof SocketException) {
407             this.uploadPolicy.displayWarn("A 'resumable' error occurred: "
408                     + errMsg);
409             // If it was the last retry, we stop here.
410             if (nbRetry >= this.uploadPolicy.getRetryMaxNumberOf()) {
411                 this.uploadPolicy.displayWarn("Too much retries (" + nbRetry
412                         + "), exiting...");
413                 throw jue;
414             }
415 
416             DialogUploadRetry dialogUploadRetry = new DialogUploadRetry(
417                     this.uploadPolicy.getContext().getFrame(), jue, nbRetry,
418                     this.uploadPolicy);
419             // The constructor returns, when the dialog is closed. Let's check
420             // the user answer:
421             if (dialogUploadRetry.isRetryValidated()) {
422                 this.uploadPolicy.displayDebug(
423                         "The user (or the timer) choosed to retry the upload",
424                         30);
425             } else {
426                 this.uploadPolicy.displayDebug(
427                         "The user refuses to retry the upload, exiting...", 30);
428                 // No retry, let's note the exception and go out
429                 throw jue;
430             }// End of resumable exceptions management.
431         } else {
432             // This exception can't be resumed. We transmit it.
433             this.uploadPolicy
434                     .displayWarn("Non resumable error occured, exiting...");
435             throw jue;
436         }
437     }

439     /**
440      * Actual execution file(s) upload. It's called by the run methods, once for
441      * all files, or file by file, depending on the UploadPolicy. The list of
442      * files to upload is stored in the packet parameter.<BR>
443      * This method is called by the run() method. The prerequisite about the
444      * filesToUpload array are: <DIR> <LI>If the sum of contentLength for the
445      * files in the array is more than the maxChunkSize, then
446      * nbFilesToUploadParam is one. <LI>The number of elements in filesToUpload
447      * is less (or equal) than the nbMaxFilesPerUpload. </DIR>
448      * 
449      * @throws JUploadException
450      * @throws JUploadInterrupted Thrown when an interruption of the thread is
451      *             detected.
452      */
453     final private void doUpload(UploadFilePacket packet)
454             throws JUploadException, JUploadInterrupted {
455         boolean bChunkEnabled = false;
456         long totalContentLength = 0;
457         long totalFileLength = 0;
458 
459         // We are about to start a new upload.
460         this.fileUploadManagerThread.setUploadStatus(packet, packet.get(0),
461                 FileUploadManagerThread.UPLOAD_STATUS_UPLOADING);
462 
463         // Prepare upload, for all files to be uploaded.
464         beforeRequest(packet);
465 
466         for (UploadFileData uploadFileData : packet) {
467             // The upload may be finished, while we're working on the files...
468             if (this.fileUploadManagerThread.isUploadFinished()) {
469                 // Let's stop our work here.
470                 return;
471             }
472             // Total length, for HTTP upload.
473             totalContentLength += uploadFileData.getUploadLength();
474             totalContentLength += getAdditionnalBytesForUpload(uploadFileData);
475             // Total file length: used to manage the progress bar (we don't
476             // follow the bytes uploaded within headers and forms).
477             totalFileLength += uploadFileData.getUploadLength();
478 
479             this.uploadPolicy.displayDebug("file "
480                     + uploadFileData.getFileName() + ": content="
481                     + uploadFileData.getUploadLength()
482                     + " bytes, getAdditionnalBytesForUpload="
483                     + getAdditionnalBytesForUpload(uploadFileData) + " bytes",
484                     50);
485         }// for
486 
487         // Ok, now we check that the totalContentLength is less than the chunk
488         // size.
489         if (totalFileLength >= this.maxChunkSize) {
490             // hum, hum, we have to download file by file, with chunk enabled.
491             // This a prerequisite of this method.
492             if (packet.size() > 1) {
493                 this.fileUploadManagerThread
494                         .setUploadException(new JUploadException(
495                                 "totalContentLength >= chunkSize: this.filesToUpload.length should be 1 (doUpload)"));
496             }
497             bChunkEnabled = true;
498         }
499 
500         // Now, we can actually do the job. This is delegate into smaller
501         // method, for easier understanding.
502         if (bChunkEnabled) {
503             // No more than one file, when in chunk mode.
504             if (packet.size() > 1) {
505                 throw new JUploadException(
506                         "totalContentLength >= chunkSize: this.filesToUpload.length should not be more than 1 (doUpload)");
507             }
508             doChunkedUpload(packet);
509         } else {
510             doNonChunkedUpload(packet, totalContentLength, totalFileLength);
511         }
512 
513         // If the request properly finished, we remove the files from the list
514         // of files to upload.
515         if (this.fileUploadManagerThread.getUploadException() == null
516                 && !this.fileUploadManagerThread.isUploadStopped()) {
517             this.fileUploadManagerThread.currentRequestIsFinished(packet);
518         }
519     }

521     /**
522      * Execution of an upload, in chunk mode. This method expects that the given
523      * packet contains only one file.
524      * 
525      * @param packet The packet that contains the file to upload in chunk mode
526      * @throws JUploadException When any error occurs, or when there is more
527      *             than one file in packet.
528      * @throws JUploadInterrupted Thrown when an interruption of the thread is
529      *             detected.
530      */
531     final private void doChunkedUpload(UploadFilePacket packet)
532             throws JUploadException, JUploadInterrupted {
533         boolean bLastChunk = false;
534         int chunkPart = 0;
535 
536         long contentLength = 0;
537         long thisChunkSize = 0;
538 
539         if (packet.size() > 1) {
540             throw new JUploadException(
541                     "doChunkedUpload called with a packet of more than 1 file ("
542                             + packet.size() + " files)");
543         }
544         UploadFileData uploadFileData = packet.get(0);
545 
546         // This while enables the chunk management:
547         // In chunk mode, it loops until the last chunk is uploaded. This works
548         // only because, in chunk mode, files are uploaded one y one (the for
549         // loop within the while loops through ... 1 unique file).
550         // In normal mode, it does nothing, as the bLastChunk is set to true in
551         // the first test, within the while.
552         while (!bLastChunk && !this.fileUploadManagerThread.isUploadFinished()) {
553             // Let's manage chunk:
554             // Files are uploaded one by one. This is checked just above.
555             chunkPart += 1;
556             bLastChunk = (contentLength > uploadFileData.getRemainingLength());
557 
558             // Is this the last chunk ?
559             if (bLastChunk) {
560                 thisChunkSize = uploadFileData.getRemainingLength();
561             } else {
562                 thisChunkSize = this.maxChunkSize;
563             }
564             contentLength = thisChunkSize
565                     + getAdditionnalBytesForUpload(uploadFileData);
566 
567             // We are about to start a new upload.
568             this.fileUploadManagerThread.setUploadStatus(packet,
569                     uploadFileData,
570                     FileUploadManagerThread.UPLOAD_STATUS_UPLOADING);
571 
572             // Ok, we've prepare the job for chunk upload. Let's do it!
573             startRequest(contentLength, true, chunkPart, bLastChunk);
574 
575             try {
576 
577                 // Let's add any file-specific header.
578                 beforeFile(packet, uploadFileData);
579 
580                 // Actual upload of the file:
581                 uploadFileData.uploadFile(getOutputStream(), thisChunkSize);
582 
583                 // Caution : this is for debug only. In production mode, sendResumableError should always be 'false'
584                 if (chunkPart == 2 && sendResumableError) {
585                     sendResumableError = false;
586                     throw new JUploadException(
587                             new SocketException(
588                                     "This is a debug error. Should not happen in production."));
589                 }
590 
591                 // If we are not in chunk mode, or if it was the last chunk,
592                 // upload should be finished.
593                 if (bLastChunk && uploadFileData.getRemainingLength() > 0) {
594                     throw new JUploadExceptionUploadFailed(
595                             "Files has not be entirely uploaded. The remaining size is "
596                                     + uploadFileData.getRemainingLength()
597                                     + " bytes. File size was: "
598                                     + uploadFileData.getUploadLength()
599                                     + " bytes.");
600 
601                 }
602                 // Let's add any file-specific header.
603                 afterFile(uploadFileData);
604 
605                 // Let's finish the request, and wait for the server Output, if
606                 // any (not applicable in FTP)
607                 int status = finishRequest();
608 
609                 if (bLastChunk) {
610                     // We are finished with this one. Let's display it.
611                     this.fileUploadManagerThread
612                             .setUploadStatus(
613                                     packet,
614                                     uploadFileData,
615                                     FileUploadManagerThread.UPLOAD_STATUS_FILE_UPLOADED_WAITING_FOR_RESPONSE);
616                 } else {
617                     // We are finished with the current chunk, but not with the
618                     // file. Let's display it.
619                     this.fileUploadManagerThread
620                             .setUploadStatus(
621                                     packet,
622                                     uploadFileData,
623                                     FileUploadManagerThread.UPLOAD_STATUS_CHUNK_UPLOADED_WAITING_FOR_RESPONSE);
624                 }
625 
626                 // We now ask to the uploadPolicy, if it was a success.
627                 // If not, the isUploadSuccessful should raise an exception.
628                 this.uploadPolicy.checkUploadSuccess(status, getResponseMsg(),
629                         getResponseBody());
630             }// try
631             finally {
632                 cleanRequest();
633             }
634         }// while
635         // Let's tell our manager that we've done the job!
636         this.fileUploadManagerThread.anotherFileHasBeenSent(packet,
637                 uploadFileData);
638 
639     }// doChunkedUpload

641     /**
642      * Execution of an upload, in standard mode. This method uploads all files
643      * in the given packet.
644      * 
645      * @param packet The files to upload in the current request to the server
646      * @param totalContentLength The total size of the upload, including any
647      *            protocol-specific header or footer.
648      * @param totalFileLength The sum of each file length.
649      * @throws JUploadException When any error occurs
650      * @throws JUploadInterrupted Thrown when an interruption of the thread is
651      *             detected.
652      */
653     final private void doNonChunkedUpload(UploadFilePacket packet,
654             final long totalContentLength, final long totalFileLength)
655             throws JUploadException, JUploadInterrupted {
656 
657         // First step is to prepare all files.
658         startRequest(totalContentLength, false, 0, true);
659 
660         try {
661 
662             // Then, upload each file.
663             for (UploadFileData uploadFileData : packet) {
664                 if (this.fileUploadManagerThread.isUploadFinished()) {
665                     // Upload is finished (by the user or because of an error,
666                     // or
667                     // instance)
668                     break;
669                 }
670                 // We are about to start a new upload.
671                 this.fileUploadManagerThread.setUploadStatus(packet,
672                         uploadFileData,
673                         FileUploadManagerThread.UPLOAD_STATUS_UPLOADING);
674 
675                 // Let's add any file-specific header.
676                 beforeFile(packet, uploadFileData);
677 
678                 // Actual upload of the file:
679                 if (!this.fileUploadManagerThread.isUploadFinished()) {
680                     uploadFileData.uploadFile(getOutputStream(), uploadFileData
681                             .getUploadLength());
682                 }
683 
684                 // Let's add any file-specific header.
685                 if (!this.fileUploadManagerThread.isUploadFinished()) {
686                     afterFile(uploadFileData);
687 
688                     // Let's tell our manager that we've done the job!
689                     // Ok, maybe the server will refuse it, but we won't say
690                     // that
691                     // now!
692                     this.fileUploadManagerThread.anotherFileHasBeenSent(packet,
693                             uploadFileData);
694                 }
695             }
696 
697             // We are finished with this one. Let's display it.
698             if (!this.fileUploadManagerThread.isUploadFinished()) {
699                 this.fileUploadManagerThread
700                         .setUploadStatus(
701                                 packet,
702                                 packet.get(packet.size() - 1),
703                                 FileUploadManagerThread.UPLOAD_STATUS_FILE_UPLOADED_WAITING_FOR_RESPONSE);
704 
705                 // Let's finish the request, and wait for the server Output, if
706                 // any (not applicable in FTP)
707                 int status = finishRequest();
708 
709                 // We now ask to the uploadPolicy, if it was a success.
710                 // If not, the isUploadSuccessful should raise an exception.
711                 this.uploadPolicy.checkUploadSuccess(status, getResponseMsg(),
712                         getResponseBody());
713             }
714         }// try
715         finally {
716             cleanRequest();
717         }
718 
719     }// doNonChunkedUpload

744     /**
745      * Replace single \r and \n by uniform end of line characters (CRLF). This
746      * makes it easier, to search for string within the body.
747      * 
748      * @param s The original string
749      * @return The string with single \r and \n modified changed to CRLF (\r\n).
750      */
751     public final String normalizeCRLF(String s) {
752         Pattern p = Pattern.compile("\\r\\n|\\r|\\n", Pattern.MULTILINE);
753         String[] lines = p.split(s);
754         // Worst case: the s string contains only \n or \r characters: we then
755         // need to triple the string length. Let's say double is enough.
756         StringBuffer sb = new StringBuffer(s.length() * 2);
757         for (int i = 0; i < lines.length; i += 1) {
758             sb.append(lines[i]).append("\r\n");
759         }
760 
761         return sb.toString();
762     }

765      * Replace \r and \n by correctly displayed end of line characters. Used to
766      * display debug output. It also replace any single \r or \n by \r\n, to
767      * make it easier, to search for string within the body.
768      * 
769      * @param s The original string
770      * @return The string with \r and \n modified, to be correctly displayed.
771      */
772     public final String quoteCRLF(String s) {
773         return s.replaceAll("\r\n", "\\\\r\\\\n\n");
