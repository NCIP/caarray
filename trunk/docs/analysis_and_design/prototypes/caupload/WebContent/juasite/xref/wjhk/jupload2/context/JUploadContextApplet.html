77  	/**
78  	 * This method checks that the applet is signed. To do this, it just check
79  	 * that the current folder is readable, and that the applet can create
80  	 * temporary files. Should be enough.<BR>
81  	 * If anyone has a better idea ... I'll be hapy to listen to it!
82  	 */
83  	void checkAppletIsSigned() {
84  
85  		// Let's be optimistic! ;-)
86  		java.lang.SecurityException ex = null;
87  		try {
88  			// I found no way to directly get the current directory (access to
89  			// user.xxx system properties is prohibited from within applets)
90  			JFileChooser fc = new JFileChooser();
91  			File currentDir = fc.getCurrentDirectory();
92  			if (!currentDir.canRead()) {
93  				ex = new java.lang.SecurityException(
94  						"The applet must be signed (can't write in '"
95  								+ currentDir.getAbsolutePath() + "')");
96  			}
97  
98  			// Let's now check that the applet may create a temporary file. May
99  			// be necessary for further processing, and for logging.
100 			try {
101 				File tmpTestFile = File.createTempFile("jupload", "test");
102 				tmpTestFile.delete();
103 			} catch (IOException ioe) {
104 				String msg = ioe.getClass().getName()
105 						+ ": Can't create temporary file (the applet is perhaps not signed (original error message: "
106 						+ ioe.getMessage() + ")";
107 				System.out.println(msg);
108 				ex = new java.lang.SecurityException(msg, ioe);
109 				ioe.printStackTrace();
110 			}
111 		} catch (java.lang.SecurityException e) {
112 			String msg = "The applet must be signed (original error message: "
113 					+ e.getMessage() + ")";
114 			System.out.println(msg);
115 			ex = new java.lang.SecurityException(msg, e);
116 		}
117 
118 		if (ex != null) {
119 			String msg = ex.getClass().getName() + " - " + ex.getMessage();
120 			JOptionPane.showMessageDialog(null, msg, "Alert",
121 					JOptionPane.ERROR_MESSAGE);
122 			System.out.println(msg);
123 			ex.printStackTrace();
124 
125 			throw ex;
126 		}
127 	}

129 	/**
130 	 * Find the JFrame which contains the applet
131 	 * 
132 	 * @param theApplet
133 	 * @return
134 	 */
135 	private Frame findParentFrame(JApplet theApplet) {
136 		Container c = theApplet;
137 		while (c != null) {
138 			if (c instanceof Frame)
139 				return (Frame) c;
140 
141 			c = c.getParent();
142 		}
143 		return (Frame) null;
144 	}

320 	/**
321 	 * @see JUploadContext#displayURL(String, boolean)
322 	 */
323 	@Override
324 	public void displayURL(String url, boolean success) {
325 		try {
326 			if (url.toLowerCase().startsWith("javascript:")) {
327 				// A JavaScript expression was specified. Execute it.
328 				String expr = url.substring(11);
329 
330 				// Replacement of %msg%. Will do something only if the %msg%
331 				// string exists in expr.
332 				expr = expr.replaceAll("%msg%", Matcher
333 						.quoteReplacement(jsString(getUploadPolicy()
334 								.getLastResponseMessage())));
335 
336 				// Replacement of %body%. Will do something only if the
337 				// %body% string exists in expr.
338 				expr = expr.replaceAll("%body%", Matcher
339 						.quoteReplacement(jsString(getUploadPolicy()
340 								.getLastResponseBody())));
341 
342 				// Replacement of %success%. Will do something only if the
343 				// %success% string exists in expr.
344 				expr = expr.replaceAll("%success%", Matcher
345 						.quoteReplacement((success) ? "true" : "false"));
346 
347 				displayDebug("Calling javascript expression: " + expr, 80);
348 				JSObject.getWindow(this.theApplet).eval(expr);
349 			} else if (success) {
350 				// This is not a javascript URL: we change the current page
351 				// only if no error occurred.
352 				String target = getUploadPolicy().getAfterUploadTarget();
353 				if (getUploadPolicy().getDebugLevel() >= 100) {
354 					getUploadPolicy().alertStr(
355 							"No switch to getAfterUploadURL, because debug level is "
356 									+ getUploadPolicy().getDebugLevel()
357 									+ " (>=100)");
358 				} else {
359 					// Let's change the current URL to edit names and
360 					// comments, for the selected album. Ok, let's go and
361 					// add names and comments to the newly updated pictures.
362 					this.theApplet.getAppletContext().showDocument(
363 							new URL(url), (null == target) ? "_self" : target);
364 				}
365 			}
366 		} catch (Exception ee) {
367 			// Oops, no navigator. We are probably in debug mode, within
368 			// eclipse for instance.
369 			try {
370 				getUploadPolicy().displayErr(ee);
371 			} catch (JUploadException e) {
372 				// Can't use standard JUpload log mode...
373 				ee.printStackTrace();
374 			}
375 		}
376 	}

378 	/**
379 	 * Generates a valid URL, from a String. The generation may add the
380 	 * documentBase of the applet.
381 	 * 
382 	 * @param url
383 	 *            A url. Can be a path relative to the current one.
384 	 * @return The normalized URL
385 	 * @throws JUploadException
386 	 */
387 	@Override
388 	public String normalizeURL(String url) throws JUploadException {
389 		if (null == url || url.length() == 0)
390 			return this.theApplet.getDocumentBase().toString();
391 		URI uri = null;
392 		try {
393 			uri = new URI(url);
394 			if (null == uri.getScheme())
395 				uri = this.theApplet.getDocumentBase().toURI().resolve(url);
396 			if (!uri.getScheme().equals("http")
397 					&& !uri.getScheme().equals("https")
398 					&& !uri.getScheme().equals("ftp")) {
399 				throw new JUploadException("URI scheme " + uri.getScheme()
400 						+ " not supported.");
401 			}
402 		} catch (URISyntaxException e) {
403 			throw new JUploadException(e);
404 		}
405 		return uri.toString();
406 	}

408 	/**
409 	 * Generate a js String, that can be written in a javascript expression.
410 	 * It's up to the caller to put the starting and ending quotes. The double
411 	 * quotes are replaced by simple quotes (to let simple quotes unchanged, as
412 	 * it may be used in common language). Thus, start and end of JS string
413 	 * should be with double quotes, when using the return of this function.
414 	 * 
415 	 * @param s
416 	 * @return The transformed string, that can be written in the output, into a
417 	 *         javascript string. It doesn't contain the starting and ending
418 	 *         double quotes.
419 	 */
420 	public String jsString(String s) {
421 		String dollarReplacement = Matcher.quoteReplacement("\\$");
422 		String singleQuoteReplacement = Matcher.quoteReplacement("\\'");
423 		String linefeedReplacement = Matcher.quoteReplacement("\\n");
424 
425 		if (s == null || s.equals("")) {
426 			return "";
427 		} else {
428 			s = s.replaceAll("\\$", dollarReplacement);
429 			s = s.replaceAll("\"", "'");
430 			s = s.replaceAll("'", singleQuoteReplacement);
431 			s = s.replaceAll("\n", linefeedReplacement);
432 			s = s.replaceAll("\r", "");
433 			return s;
434 		}
