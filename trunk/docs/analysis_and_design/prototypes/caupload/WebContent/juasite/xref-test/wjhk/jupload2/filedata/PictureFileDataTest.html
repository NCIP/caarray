236             if (charTest < 0) {

240                 isTest.close();
241                 isAssert.close();
242                 return;
243             }
244             if (charTest != charAssert) {

247                 // stop here.
248                 isTest.close();
249                 isAssert.close();
250                 return;
251             }
252         }// while
253         charTest = isTest.read();
254         if (charTest >= 0) {
255             // There is a difference between the two files: the file has
256             // been transformed. We can't really test more here, so let's
257             // stop here.
258             isTest.close();
259             isAssert.close();
260             return;
261         }
262         isTest.close();
263         isAssert.close();
264 
265         Assert
266                 .fail("The files are identifical (although the pictureFileData should have transformed it !");

396                     "The file should be removed from the file system", tempFile
397                             .canRead());
398         } finally {
399             if (tempFile != null) {

402             }
403         }
