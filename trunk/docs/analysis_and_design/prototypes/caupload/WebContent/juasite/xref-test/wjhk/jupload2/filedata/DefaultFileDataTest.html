131 	@Test
132 	public void testAppendFileProperties() throws JUploadIOException {
133 		// /////////////////////////////////////////////////////////////////////////////
134 		// ////////// A Utility class, to check the properties wich are managed
135 		// /////////////////////////////////////////////////////////////////////////////
136 		class ByteArrayEncoderHTTP_CheckAppendTextProperty extends
137 				ByteArrayEncoderHTTP {
138 			String mimeType = null;
139 
140 			String pathInfo = null;
141 
142 			String relpathinfo = null;
143 
144 			String filemodificationdate = null;
145 
146 			ByteArrayEncoderHTTP_CheckAppendTextProperty()
147 					throws JUploadIOException {
148 				super(DefaultFileDataTest.this.uploadPolicy);
149 			}
150 
151 			@Override
152 			public ByteArrayEncoder appendTextProperty(String name,
153 					String value, int index) throws JUploadIOException {
154 				if (name.equals("mimetype")) {
155 					this.mimeType = value;
156 				} else if (name.equals("pathinfo")) {
157 					this.pathInfo = value;
158 				} else if (name.equals("relpathinfo")) {
159 					this.relpathinfo = value;
160 				} else if (name.equals("filemodificationdate")) {
161 					this.filemodificationdate = value;
162 				} else {
163 					throw new java.lang.IllegalArgumentException(
164 							"Unknown property : " + name);
165 				}
166 				return this;
167 			}
168 		}
169 		// /////////////////////////////////////////////////////////////////////////////
170 		ByteArrayEncoderHTTP_CheckAppendTextProperty bae = new ByteArrayEncoderHTTP_CheckAppendTextProperty();
171 
172 		int index = 58;
173 		this.fileData.appendFileProperties(bae, index);
174 		Assert.assertEquals("Check mimeType value",
175 				JUploadContextTestHelper.TEST_CASE_MIME_TYPE, bae.mimeType);
176 		Assert.assertEquals("Check pathInfo value", this.file.getParentFile()
177 				.getAbsolutePath(), bae.pathInfo);
178 
179 		// The relpathinfo, is the path relative to the file root.
180 		Assert
181 				.assertEquals(
182 						"The relpathinfo is the part of the file absolute path, wich is after the file root",
183 						this.file.getParentFile().getAbsolutePath(), this.root
184 								.getAbsolutePath()
185 								+ File.separator + bae.relpathinfo);
186 		// date here.
187 		Assert.assertNotNull("Check mimeType value", bae.filemodificationdate);
188 	}
