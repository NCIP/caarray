155 		int maxNbFiles = 1000;
156 		int realMaxNbFiles = Math.min(maxNbFiles, NB_FILES_IN_FILE_PANEL);
157 		int maxNbBytes = 100;
158 
159 		System.setProperty(UploadPolicy.PROP_NB_FILES_PER_REQUEST, String
160 				.valueOf(maxNbFiles));
161 		System.setProperty(UploadPolicy.PROP_MAX_CHUNK_SIZE, String
162 				.valueOf(maxNbBytes));
163 		// We need to re-initialize this test, to use the above setting.
164 		createTestData();
165 		Assert.assertEquals("The maximal number of files", realMaxNbFiles,
166 				this.uploadFilePacket.maxNbFiles);
167 		Assert.assertEquals("The maximal number of bytes", maxNbBytes,
168 				this.uploadFilePacket.maxNbBytes);
169 
170 		// The packet is full only if we have exactly the good number of bytes !
171 		String fileContent = "a";
172 		for (int i = 1; i < maxNbBytes; i += 1) {
173 			this.fileData = new FileDataTestHelper(0);
174 			((FileDataTestHelper) this.fileData).fileContent = fileContent;
175 			this.fileData.beforeUpload();
176 			Assert.assertTrue("The file must be accepted",
177 					this.uploadFilePacket
178 							.add(new UploadFileData(this.fileData, 0,
179 									this.fileUploadManagerThread,
180 									this.uploadPolicy)));
181 			Assert.assertFalse("This packet is not full yet (" + i + " files)",
182 					this.uploadFilePacket.isFull());
183 		}
184 
185 		this.fileData = new FileDataTestHelper(0);
186 		((FileDataTestHelper) this.fileData).fileContent = fileContent;
187 		this.uploadFilePacket.add(new UploadFileData(this.fileData, 0,
188 				this.fileUploadManagerThread, this.uploadPolicy));
189 		Assert.assertTrue("This packet is now full", this.uploadFilePacket
190 				.isFull());
191 	}

210 		for (int i = 0; i < nbLoop; i += 1) {
211 			fileData = new FileDataTestHelper(0);
212 			fileData.beforeUpload();
213 			Assert.assertTrue("The file must be accepted",
214 					this.uploadFilePacket
215 							.canAdd(new UploadFileData(fileData, 0,
216 									this.fileUploadManagerThread,
217 									this.uploadPolicy)));
218 			Assert.assertTrue("The file must be accepted",
219 					this.uploadFilePacket.add(new UploadFileData(fileData, 0,
220 							this.fileUploadManagerThread, this.uploadPolicy)));
221 		}
222 
223 		fileData = new FileDataTestHelper(0);
224 		Assert.assertFalse("The file must not be accepted",
225 				this.uploadFilePacket.canAdd(new UploadFileData(fileData, 0,
226 						this.fileUploadManagerThread, this.uploadPolicy)));
227 		Assert.assertFalse("The file must not be accepted",
228 				this.uploadFilePacket.add(new UploadFileData(fileData, 0,
229 						this.fileUploadManagerThread, this.uploadPolicy)));
