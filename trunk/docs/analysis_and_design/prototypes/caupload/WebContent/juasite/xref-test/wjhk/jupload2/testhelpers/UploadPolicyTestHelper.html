569 	/**
570 	 * @see wjhk.jupload2.policies.UploadPolicy#getSslVerifyCert()
571 	 */
572 	public int getSslVerifyCert() {
573 		final String sslVerifyCert = this.uploadContext.getParameter(
574 				PROP_SSL_VERIFY_CERT, DEFAULT_SSL_VERIFY_CERT);
575 		if (sslVerifyCert.toLowerCase().equals("none")) {
576 			return InteractiveTrustManager.NONE;
577 		} else if (sslVerifyCert.toLowerCase().equals("server")) {
578 			return InteractiveTrustManager.SERVER;
579 		} else if (sslVerifyCert.toLowerCase().equals("client")) {
580 			return InteractiveTrustManager.CLIENT;
581 		} else if (sslVerifyCert.toLowerCase().equals("strict")) {
582 			return InteractiveTrustManager.STRICT;
583 		} else {
584 			this.logger.warn("Invalid parameter sslVerifyCert ("
585 					+ sslVerifyCert + ")");
586 			return InteractiveTrustManager.NONE;
587 		}
588 	}
