166     /**
167      * This method tries to determine if the current computer can access to
168      * jupload.sourceforge.net. If yes, the
169      * http://jupload.sourceforge.net/upload_dummy.html URL is used. <BR>
170      * If no, the http://localhost/index.html URL is used for test postURL.<BR>
171      * <BR>
172      * The reason for this test, is that I often work unconnected, in the train.
173      * And I still want the unit tests to work properly.
174      */
175     private void setPostURL() {
176         // If the system property has not been set yet, let's determine if
177         // we're connected to the network.
178         if (!postURLHasBeenSet) {
179             String postURL;
180             try {
181                 URL url = new URL(DEFAULT_POST_URL);
182                 new Socket(url.getHost(), url.getPort());
183                 // The given host is valid. We use this URL.
184                 postURL = DEFAULT_POST_URL;
185             } catch (Exception e) {
186                 logger.warn(e.getClass().getName() + " when creating URL from "
187                         + DEFAULT_POST_URL + " (will use "
188                         + DEFAULT_LOCAL_POST_URL + " instead");
189                 //
190                 postURL = DEFAULT_LOCAL_POST_URL;
191             }
192 
193             System.setProperty(UploadPolicy.PROP_POST_URL, postURL);
194             postURLHasBeenSet = true;
195         }
196     }

198     /**
199      * @param nbFiles
200      */
201     public void setupFileList(int nbFiles) {
202 
203         this.filesToUpload = new ArrayList<FileData>(nbFiles);
204         File[] fArray = new File[nbFiles];
205 
206         for (int i = 0; i < nbFiles; i += 1) {
207             FileData fileData = new FileDataTestHelper(i);
208             // We must be able to load the file. Otherwise, it's useless to
209             // start. And there seems to be problem with user dir, depending on
210             // the java tool used.
211             Assert.assertTrue(fileData.getFileName() + " must be readable !",
212                     fileData.canRead());
213             this.filesToUpload.add(fileData);
214         }
215 
216         // Let's add these files to the FilePanel
217         if (this.filePanel instanceof FilePanelTestHelper) {
218             ((FilePanelTestHelper) this.filePanel).filesToUpload = this.filesToUpload;
219         } else {
220             // We first clear the list, to be sure of the final content.
221             this.filePanel.removeAll();
222             this.filePanel.addFiles(fArray, null);
223         }
224     }

237     /**
238      * Wait for a queue to be emptied by a consuming thread. This will wait
239      * {@link #MAX_WAIT_FOR_THREAD} at most, and check this every
240      * {@link INTERVAL_BEFORE_CHECKING_THREAD} ms.
241      */
242     @SuppressWarnings("unchecked")
243     void waitForQueueToBeEmpty(Queue queue, String queueName) {
244         int nbLoop = MAX_WAIT_FOR_THREAD / INTERVAL_BEFORE_CHECKING_THREAD;
245         try {
246             for (int i = 0; i < nbLoop; i += 1) {
247                 if (queue.isEmpty()) {
248                     logger.info("The queue " + queueName + " has been emptied");
249                     return;
250                 }
251                 Thread.sleep(10);
252             }
253         } catch (InterruptedException e) {
254             logger.warn("waitForQueueToBeEmpty got interrupted");
255         }
256         logger.warn("The queue " + queueName + " was not emptied");
257     }
258 
259     /**
260      * Wait for a thread to finish normally. This will wait
261      * {@link #MAX_WAIT_FOR_THREAD} at most, and check this every
262      * {@link INTERVAL_BEFORE_CHECKING_THREAD} ms.
263      */
264     void waitForThreadToFinish(Thread thread, String threadName) {
265         if (thread.isAlive()) {
266             // Let's wait a little for this thread to finish...
267             try {
268                 (thread).join(MAX_WAIT_FOR_THREAD);
269             } catch (InterruptedException e) {
270                 Assert.fail("Was interrupted during the join: the thread "
271                         + threadName + " did not finish on time ");
272             }
273         }
274 
275         // Is the thread finished now ?
276         if (thread.isAlive()) {
277             logger.warn("The thread " + threadName
278                     + " did not finished alone: let's interrupt it");
279             thread.interrupt();
280         } else {
281             logger.info("The thread " + threadName
282                     + " finished alone (no interruption needed)");
283         }
284     }

286     /** */
287     @After
288     public void cleanQueues() {
289         logger
290                 .debug("Finishing the test: interrupting the threads, and cleaning the queues");
291 
292         if (this.fileUploadThread != null) {
293             if (this.fileUploadThread.isAlive()) {
294                 this.fileUploadThread.interrupt();
295                 this.fileUploadThread = null;
296             }
297         }
298         if (this.fileUploadManagerThread != null) {
299             if (this.fileUploadManagerThread.isAlive()) {
300                 this.fileUploadManagerThread.interrupt();
301                 this.fileUploadManagerThread = null;
302             }
303         }
304         if (this.packetConstructionThread != null) {
305             if (this.packetConstructionThread.isAlive()) {
306                 this.packetConstructionThread.interrupt();
307                 this.packetConstructionThread = null;
308             }
309         }
310 
311         if (this.preparedFileQueue != null) {
312             while (!this.preparedFileQueue.isEmpty()) {
313                 this.preparedFileQueue.poll();
314             }
315             this.preparedFileQueue = null;
316         }
317 
318         if (this.packetQueue != null) {
319             while (!this.packetQueue.isEmpty()) {
320                 this.packetQueue.poll();
321             }
322             this.packetQueue = null;
323         }
324     }

381     /**
382      * This static class computes create a file instance for the given relative
383      * path.<BR>
384      * This method is taken from <A HREF=
385      * "http://kozelka.net/blog/generating-temporary-files-in-junit-tests">Petr
386      * Kozelka's blog</A>
387      * 
388      * @param relativeFilePath The path of the file, starting from
389      *            src/test/resources.
390      * @return The File for the file, whose relative path to tests folder is
391      *         given in argument. This method is compatible with whatever tool
392      *         is used to execute the tests: it can be from maven, from eclipse,
393      *         or from any other tool.
394      */
395     public static File getTestFile(String relativeFilePath) {
396         final String clsUri = AbstractJUploadTestHelper.class.getName()
397                 .replace('.', File.separatorChar)
398                 + ".class";
399         final URL url = AbstractJUploadTestHelper.class.getClassLoader()
400                 .getResource(clsUri);
401         final String clsPath = url.getPath().replaceAll("%20", " ").replaceAll(
402                 "%5c", ".");
403         final File root = new File(clsPath.substring(0, clsPath.length()
404                 - clsUri.length()));
405         return new File(root, relativeFilePath);
406     }
